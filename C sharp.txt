C++:

#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	double a;
	cout << "Please enter A : ";
	cin >> a;

	cout << endl;

	double b;
	cout << "Please enter B : ";
	cin >> b;

	double x = ((a + b - abs(a - b)) / 4);
	double y = 3 * x + 5;

	cout << "X : " << x << endl;
	cout << "Y : " << y << endl;
}

C#:

Console.WriteLine("Please enter A :");
